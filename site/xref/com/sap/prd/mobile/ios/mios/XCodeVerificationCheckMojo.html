91  /**
92   * Provides the possibility to perform verification checks.<br>
93   * The check classes and their severities are described in an additional xml document, defined in
94   * <code>xcode.verification.checks.definitionFile</code>.<br>
95   * The specific checks have to be implemented in separate projects. These projects define dependency
96   * to Xcode Maven Pugin Verification API and must not reference the xcode-maven-plugin project.
97   * The Xcode Maven Plugin Verification API project could be found <a href=https://github.com/sap-production/xcode-maven-plugin-verification-api>here</a>
98   * The coordinates of that projects need to be provided on the
99   * <code>check</code> node belonging to the test as attributes <code>groupId</code>,
100  * <code>artifactId</code> and <code>version</code>.<br>
101  * The classpath for this goal will be extended by the jars found under the specified GAVs. <br>
102  * Example checks definition:
103  * 
104  * <pre>
105  * &lt;checks&gt;
106  *   &lt;check groupId="my.group.id" artifactId="artifactId" version="1.0.0" severity="ERROR" class="com.my.MyVerificationCheck1"/&gt;
107  *   &lt;check groupId="my.group.id" artifactId="artifactId" version="1.0.0" severity="WARNING" class="com.my.MyVerificationCheck2"/&gt;
108  * &lt;/checks&gt;
109  * </pre>
110  * 
111  * @goal verification-check
112  * 
113  */
114 public class XCodeVerificationCheckMojo extends BuildContextAwareMojo
115 {

119   private enum Protocol
120   {
121 
122     HTTP() {
123 
124       @Override
125       Reader getCheckDefinitions(String location) throws IOException
126       {
127         HttpClient httpClient = new DefaultHttpClient();
128         HttpGet get = new HttpGet(getName() + COLON + DOUBLE_SLASH + location);
129 
130         String response = httpClient.execute(get, new BasicResponseHandler());
131         return new StringReader(response);
132       }
133 
134     },
135     HTTPS() {
136 
137       @Override
138       Reader getCheckDefinitions(String location) throws IOException
139       {
140         HttpClient httpClient = new DefaultHttpClient();
141         try {
142           SSLContext sslcontext = SSLContext.getInstance("TLS");
143           X509TrustManager trustManager = new X509TrustManager() {
144 
145             @Override
146             public X509Certificate[] getAcceptedIssuers()
147             {
148               return null;
149             }
150 
151             @Override
152             public void checkServerTrusted(X509Certificate[] chain, String authType) throws CertificateException
153             {
154             }
155 
156             @Override
157             public void checkClientTrusted(X509Certificate[] chain, String authType) throws CertificateException
158             {
159             }
160           };
161           X509HostnameVerifier hostNameVerifier = new X509HostnameVerifier() {
162 
163             @Override
164             public boolean verify(String arg0, SSLSession arg1)
165             {
166               return true;
167             }
168 
169             @Override
170             public void verify(String host, String[] cns, String[] subjectAlts) throws SSLException
171             {
172             }
173 
174             @Override
175             public void verify(String host, X509Certificate cert) throws SSLException
176             {
177             }
178 
179             @Override
180             public void verify(String host, SSLSocket ssl) throws IOException
181             {
182             }
183           };
184 
185           final int port = new URL(getName() + COLON + DOUBLE_SLASH + location).getPort();
186           sslcontext.init(null, new TrustManager[] { trustManager }, null);
187           SSLSocketFactory sslSocketFactory = new SSLSocketFactory(sslcontext);
188           sslSocketFactory.setHostnameVerifier(hostNameVerifier);
189           ClientConnectionManager clientConnectionManager = httpClient.getConnectionManager();
190           SchemeRegistry sr = clientConnectionManager.getSchemeRegistry();
191           sr.register(new Scheme(getName(), sslSocketFactory, port));
192         }
193         catch (NoSuchAlgorithmException e) {
194           e.printStackTrace();
195         }
196         catch (KeyManagementException e) {
197           // TODO Auto-generated catch block
198           e.printStackTrace();
199         }
200         HttpGet get = new HttpGet(getName() + COLON + DOUBLE_SLASH + location);
201 
202         String response = httpClient.execute(get, new BasicResponseHandler());
203         return new StringReader(response);
204       }
205     },
206     FILE() {
207 
208       @Override
209       Reader getCheckDefinitions(String location) throws IOException
210       {
211         if (location.startsWith(DOUBLE_SLASH)) location = location.substring(DOUBLE_SLASH.length());
212         final File f = new File(location);
213         if (!f.canRead()) {
214           throw new IOException("Cannot read checkDefintionFile '" + f + "'.");
215         }
216 
217         return new InputStreamReader((new FileInputStream(f)), "UTF-8");
218       }
219     };
220     abstract Reader getCheckDefinitions(String location) throws IOException;
221 
222     String getName()
223     {
224       return name().toLowerCase(Locale.ENGLISH);
225     }

227     static String getProtocols()
228     {
229       final StringBuilder sb = new StringBuilder(16);
230       for (Protocol p : Protocol.values()) {
231         if (sb.length() != 0)
232           sb.append(", ");
233         sb.append(p.getName());
234       }
235       return sb.toString();
236     }

243       catch (final IllegalArgumentException ex) {
244         throw new InvalidProtocolException(protocol, ex);
245       }
246     }

303   /**
304    * The location where the check definition file is present. Could be a file on the local file
305    * system or a remote located file, accessed via http or https. <br>
306    * Examples:
307    * <ul>
308    * <li>-Dxcode.verification.checks.definitionFile=file:./checkDefinitionFile.xml
309    * <li>-Dxcode.verification.checks.definitionFile=http://example.com/checkDefinitionFile.xml
310    * <li>-Dxcode.verification.checks.definitionFile=https://example.com/checkDefinitionFile.xml
311    * </ul>
312    * 
313    * @parameter expression="${xcode.verification.checks.definitionFile}"
314    * @since 1.9.3
315    */
316   private String checkDefinitionFile;

318   @Override
319   public void execute() throws MojoExecutionException, MojoFailureException
320   {
321 
322     if (skip) {
323 
324       getLog()
325         .info(
326               String
327                 .format(
328                       "Verification check goal has been skipped intentionally since parameter 'xcode.verification.checks.skip' is '%s'.",
329                       skip));
330       return;
331     }
332 
333     try {
334       PackagingType.getByMavenType(packaging);
335     }
336     catch (PackagingType.UnknownPackagingTypeException ex)
337     {
338       getLog().info(
339             "Packaging type is " + packaging
340                   + ". There is no need to apply verification checks for this packaging type.");
341       return;
342     }
343 
344     try {
345 
346       final Checks checks = getChecks(checkDefinitionFile);
347 
348       if (checks.getCheck().isEmpty()) {
349         getLog().warn(String.format("No checks configured in '%s'.", checkDefinitionFile));
350       }
351 
352       Map<Check, Exception> failedChecks = new HashMap<Check, Exception>();
353 
354       for (Check check : checks.getCheck()) {
355         try {
356           final ClassRealm verificationCheckRealm = extendClasspath(check);
357           final Exception ex = performCheck(verificationCheckRealm, check);
358           if (ex != null)
359           {
360             failedChecks.put(check, ex);
361           }
362         }
363         catch (DuplicateRealmException ex) {
364           throw new MojoExecutionException(ex.getMessage(), ex);
365         }
366       }
367 
368       handleExceptions(failedChecks);
369 
370     }
371     catch (XCodeException e) {
372       throw new MojoExecutionException(e.getMessage(), e);
373     }
374     catch (IOException e) {
375       throw new MojoExecutionException(e.getMessage(), e);
376     }
377     catch (JAXBException e) {
378       throw new MojoExecutionException(e.getMessage(), e);
379     }
380     catch (DependencyCollectionException e) {
381       throw new MojoExecutionException(e.getMessage(), e);
382     }
383   }

385   private Exception performCheck(ClassRealm verificationCheckRealm, final Check checkDesc)
386         throws MojoExecutionException
387   {
388 
389     getLog().info(String.format("Performing verification check '%s'.", checkDesc.getClazz()));
390 
391     if (getLog().isDebugEnabled()) {
392 
393       final Charset defaultCharset = Charset.defaultCharset();
394       final ByteArrayOutputStream byteOs = new ByteArrayOutputStream();
395       final PrintStream ps;
396       try {
397        ps = new PrintStream(byteOs, true, defaultCharset.name());
398       } catch(UnsupportedEncodingException ex) {
399         throw new MojoExecutionException(String.format("Charset '%s' cannot be found.", defaultCharset.name()));
400       }
401       
402       try {
403         verificationCheckRealm.display(ps);
404         ps.close();
405         getLog().debug(
406               String.format("Using classloader for loading verification check '%s':%s%s", checkDesc.getClazz(),
407                     System.getProperty("line.separator"), new String(byteOs.toByteArray(), defaultCharset)));
408       }
409       finally {
410         IOUtils.closeQuietly(ps);
411       }
412     }
413 
414     try {
415       final Class<?> verificationCheckClass = Class.forName(checkDesc.getClazz(), true, verificationCheckRealm);
416 
417       getLog().debug(
418             String.format("Verification check class %s has been loaded by %s.", verificationCheckClass.getName(),
419                   verificationCheckClass.getClassLoader()));
420       getLog().debug(
421             String.format("Verification check super class %s has been loaded by %s.", verificationCheckClass
422               .getSuperclass().getName(), verificationCheckClass.getSuperclass().getClassLoader()));
423       getLog().debug(
424             String.format("%s class used by this class (%s) has been loaded by %s.", VerificationCheck.class.getName(),
425                   this.getClass().getName(), VerificationCheck.class.getClassLoader()));
426 
427       for (final String configuration : getConfigurations()) {
428         for (final String sdk : getSDKs()) {
429           getLog().info(
430                 String.format("Executing verification check: '%s' for configuration '%s' and sdk '%s'.",
431                       verificationCheckClass.getName(), configuration, sdk));
432           final VerificationCheck verificationCheck = (VerificationCheck) verificationCheckClass.newInstance();
433           verificationCheck.setXcodeContext(getXCodeContext(SourceCodeLocation.WORKING_COPY, configuration, sdk));
434           verificationCheck.setMavenProject(project);
435           verificationCheck.setEffectiveBuildSettings(new EffectiveBuildSettings());
436           try {
437             verificationCheck.check();
438           }
439           catch (VerificationException ex) {
440             return ex;
441           }
442           catch (RuntimeException ex) {
443             return ex;
444           }
445         }
446       }
447       return null;
448     }
449     catch (ClassNotFoundException ex) {
450       throw new MojoExecutionException(
451             "Could not load verification check '"
452                   + checkDesc.getClazz()
453                   + "'. May be your classpath has not been properly extended. "
454                   +
455                   "Provide the GAV of the project containing the check as attributes as part of the check defintion in the check configuration file.",
456             ex);
457     }
458     catch (NoClassDefFoundError err) {
459       getLog().error(String.format("Could not load verification check '%s'. " +
460             "May be your classpath has not been properly extended. " +
461             "Additional dependencies need to be declard inside the check definition file: %s",
462             checkDesc.getClazz(), err.getMessage()), err);
463       throw err;
464     }
465     catch (InstantiationException ex) {
466       throw new MojoExecutionException(String.format("Could not instanciate verification check '%s': %s",
467             checkDesc.getClazz(), ex.getMessage()), ex);
468     }
469     catch (IllegalAccessException ex) {
470       throw new MojoExecutionException(String.format("Could not access verification check '%s': %s",
471             checkDesc.getClazz(), ex.getMessage()), ex);
472     }
473 
474   }

476   private void handleExceptions(Map<Check, Exception> failedChecks)
477         throws MojoExecutionException
478   {
479     boolean mustFailedTheBuild = false;
480     for (Map.Entry<Check, Exception> entry : failedChecks.entrySet()) {
481       handleException(entry.getKey(), entry.getValue());
482       if (entry.getKey().getSeverity().equalsIgnoreCase("ERROR")) {
483         mustFailedTheBuild = true;
484       }
485     }
486     if (mustFailedTheBuild) {
487       throw new MojoExecutionException("Verification checks failed. See the log file for details.");
488     }
489   }
490 
491   private void handleException(Check failedCheck, final Exception e)
492   {
493     final String message;
494     if (e instanceof VerificationException) {
495       message = "Verification check '" + failedCheck.getClazz() + " failed. " + e.getMessage();
496     }
497     else {
498       message = "Cannot perform check: " + failedCheck.getClazz() + ". Error during test setup " + e.getMessage();
499     }
500     if (failedCheck.getSeverity().equalsIgnoreCase("WARNING")) {
501       getLog().warn(message);
502     }
503     else {
504       getLog().error(message);
505     }
506   }

508   private ClassRealm extendClasspath(Check check) throws XCodeException, DependencyCollectionException,
509         DuplicateRealmException, MalformedURLException
510   {
511     final org.sonatype.aether.artifact.Artifact artifact = parseDependency(check);
512 
513     final ClassLoader loader = this.getClass().getClassLoader();
514 
515     if (!(loader instanceof ClassRealm)) {
516 
517       throw new XCodeException("Could not add jar to classpath. Class loader '" + loader
518             + "' is not an instance of '" + ClassRealm.class.getName() + "'.");
519     }
520 
521     final ClassRealm classRealm = (ClassRealm) loader;
522 
523     if (artifact == null)
524     {
525       return classRealm;
526     }
527 
528     final Set<String> scopes = new HashSet<String>(Arrays.asList(org.apache.maven.artifact.Artifact.SCOPE_COMPILE,
529           org.apache.maven.artifact.Artifact.SCOPE_PROVIDED,
530           org.apache.maven.artifact.Artifact.SCOPE_RUNTIME,
531           org.apache.maven.artifact.Artifact.SCOPE_SYSTEM)); // do not resolve dependencies with scope "test".
532 
533     final XCodeDownloadManager downloadManager = new XCodeDownloadManager(projectRepos, repoSystem, repoSession);
534 
535     final Set<org.sonatype.aether.artifact.Artifact> theEmptyOmitsSet = Collections.emptySet();
536     final Set<org.sonatype.aether.artifact.Artifact> omits = downloadManager.resolveArtifactWithTransitveDependencies(
537           new Dependency(getVerificationAPIGav(), org.apache.maven.artifact.Artifact.SCOPE_COMPILE), scopes,
538           theEmptyOmitsSet);
539 
540     omits.add(getVerificationAPIGav());
541 
542     final Set<org.sonatype.aether.artifact.Artifact> artifacts = downloadManager
543       .resolveArtifactWithTransitveDependencies(new Dependency(artifact,
544             org.apache.maven.artifact.Artifact.SCOPE_COMPILE), scopes, omits);
545 
546     final ClassRealm childClassRealm = classRealm.createChildRealm(getUniqueRealmId(classRealm.getWorld(),
547           classRealm.getId() + "-" + check.getClazz()));
548 
549     addDependencies(childClassRealm, artifacts);
550 
551     return childClassRealm;
552   }

554   private String getUniqueRealmId(final ClassWorld world, final String realmIdPrefix)
555   {
556     String uniqueRealmIdCandidate = null;
557     int i = 0;
558     while (true) {
559       uniqueRealmIdCandidate = realmIdPrefix + "-" + i;
560       if (world.getClassRealm(uniqueRealmIdCandidate) == null)
561       {
562         return uniqueRealmIdCandidate;
563       }
564       i++;
565     }
566   }

568   private void addDependencies(final ClassRealm childClassRealm, Set<org.sonatype.aether.artifact.Artifact> artifacts)
569         throws MalformedURLException
570   {
571     for (org.sonatype.aether.artifact.Artifact a : artifacts)
572     {
573       childClassRealm.addURL(a.getFile().toURI().toURL());
574     }

577   static org.sonatype.aether.artifact.Artifact parseDependency(final Check check)
578         throws XCodeException
579   {
580     final String groupId = check.getGroupId();
581     final String artifactId = check.getArtifactId();
582     final String version = check.getVersion();
583 
584     if (StringUtils.isEmpty(groupId) && StringUtils.isEmpty(artifactId) && StringUtils.isEmpty(version)) {
585       log.info(
586         "No coordinates maintained for check represented by class '" + check.getClazz()
587               + "'. Assuming this check is already contained in the classpath.");
588       return null;
589     }
590 
591     if (StringUtils.isEmpty(groupId))
592       throw new XCodeException(String.format("groupId for check %s is null or emtpy", check.getClazz()));
593 
594     if (StringUtils.isEmpty(artifactId))
595       throw new XCodeException(String.format("artifactId for check %s is null or emtpy", check.getClazz()));
596 
597     if (StringUtils.isEmpty(version))
598       throw new XCodeException(String.format("version for check %s is null or emtpy", check.getClazz()));
599 
600     return new DefaultArtifact(groupId, artifactId, "jar", version);
601   }

603   static Checks getChecks(final String checkDefinitionFileLocation) throws XCodeException, IOException, JAXBException
604   {
605     Reader checkDefinitions = null;
606 
607     try {
608       checkDefinitions = getChecksDescriptor(checkDefinitionFileLocation);
609       return (Checks) JAXBContext.newInstance(Checks.class).createUnmarshaller().unmarshal(checkDefinitions);
610     }
611     finally {
612       IOUtils.closeQuietly(checkDefinitions);
613     }
614   }

616   org.sonatype.aether.artifact.Artifact getVerificationAPIGav() throws XCodeException
617   {
618 
619     InputStream is = null;
620 
621     try {
622       is = XCodeVerificationCheckMojo.class.getResourceAsStream("/misc/project.properties");
623 
624       if (is == null)
625       {
626         throw new XCodeException("Cannot get the GAV of the xcode-maven-plugin");
627       }
628 
629       Properties props = new Properties();
630       props.load(is);
631 
632       final String groupId = props.getProperty("verification.api.groupId");
633       final String artifactId = props.getProperty("verification.api.artifactId");
634       final String version = props.getProperty("verification.api.version");
635       return new DefaultArtifact(groupId, artifactId, "jar", version);
636     }
637     catch (final IOException ex) {
638       throw new XCodeException("Cannot get the GAV for the verification API", ex);
639     }
640     finally {
641       IOUtils.closeQuietly(is);
642     }
643   }

645   static Reader getChecksDescriptor(final String checkDefinitionFileLocation) throws XCodeException, IOException
646   {
647     if (checkDefinitionFileLocation == null || checkDefinitionFileLocation.trim().isEmpty()) {
648       throw new XCodeException(
649             "CheckDefinitionFile was not configured. Cannot perform verification checks. Define check definition file with paramater 'xcode.verification.checks.definitionFile'.");
650     }
651 
652     final Location location = Location.getLocation(checkDefinitionFileLocation);
653 
654     try {
655       Protocol protocol = Protocol.valueOf(location.protocol);
656       return protocol.getCheckDefinitions(location.location);
657     }
658     catch (IllegalArgumentException ex) {
659       throw new InvalidProtocolException(format("Invalid protocol provided: '%s'. Supported values are:'%s'.",
660             location.protocol, Protocol.getProtocols()), ex);
661     }
662     catch (IOException ex) {
663       throw new IOException(format("Cannot get check definitions from '%s'.", checkDefinitionFileLocation), ex);
664     }
665   }

667   static class Location
668   {
669     static Location getLocation(final String locationUriString) throws InvalidProtocolException, NoProtocolException,
670           MalformedURLException
671     {
672       final URL url;
673 
674       try {
675         url = new URL(locationUriString.trim());
676       }
677       catch (MalformedURLException ex) {
678 
679         //
680         // trouble with protocol ???
681         //
682 
683         try {
684 
685           if (URI.create(locationUriString).getScheme() == null)
686           {
687             throw new NoProtocolException(String.format(
688                   "Provide a protocol [%s] for parameter 'xcode.verification.checks.definitionFile'",
689                   Protocol.getProtocols()), ex);
690           }
691         }
692         catch (RuntimeException ignore) {
693           //
