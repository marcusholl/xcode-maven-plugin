84    void prepareBuild(final MavenProject project, Set<String> configurations,
85          final Set<String> sdks) throws MojoExecutionException, XCodeException, IOException
86    {
87  
88      prepareRootFolders(project, configurations, sdks);
89  
90      final Iterator<Artifact> dependentArtifacts = project.getArtifacts().iterator();
91  
92      if (!dependentArtifacts.hasNext()) {
93        LOGGER.info("No dependencies found.");
94      }
95  
96      while (dependentArtifacts.hasNext()) {
97  
98        final Artifact mainArtifact = (Artifact) dependentArtifacts.next();
99  
100       LOGGER.info("Preparing dependency: " + mainArtifact.getId());
101 
102       if (PackagingType.LIB.getMavenPackaging().equals(mainArtifact.getType())) {
103         prepareLibrary(project, configurations, sdks, mainArtifact);
104       }
105       else if (PackagingType.FRAMEWORK.getMavenPackaging().equals(mainArtifact.getType())) {
106         prepareFramework(project, mainArtifact, configurations);
107       }
108       else if (additionalPackagingTypes.keySet().contains(mainArtifact.getType())) {
109 
110         final PackagingTypeAction packagingTypeAction = PackagingTypeAction.valueOf(additionalPackagingTypes
111           .get(mainArtifact.getType()));
112         LOGGER.info("Packaging type '" + mainArtifact.getType() + "' found in pom. Action: " + packagingTypeAction);
113         packagingTypeAction.perform(archiverManager, project, mainArtifact);
114       }
115       else {
116 
117         LOGGER.warning("Unknown dependency type detected: '" + mainArtifact.getType() + "'. The corresponding dependency '"
118               + mainArtifact.getGroupId() + ":" + mainArtifact.getArtifactId() + ":" + mainArtifact.getVersion()
119               + "' will be ignored.");
120       }
121     }
122   }

124   private void prepareLibrary(final MavenProject project,
125         Set<String> configurations, final Set<String> sdks,
126         final Artifact mainArtifact) throws MojoExecutionException,
127         IOException, XCodeException
128   {
129     for (final String xcodeConfiguration : configurations) {
130 
131       Map<String, File> thinLibs = new HashMap<String, File>();
132 
133       for (final String sdk : sdks) {
134 
135         thinLibs.put(sdk, resolveThinLib(project, xcodeConfiguration, sdk, mainArtifact));
136 
137         try {
138           prepareHeaders(project, xcodeConfiguration, sdk, mainArtifact);
139         }
140         catch (SideArtifactNotFoundException e) {
141           LOGGER.info("Headers not found for: '" + mainArtifact.getGroupId() + ":" + mainArtifact.getArtifactId()
142                 + ":"
143                 + mainArtifact.getVersion() + ":" + mainArtifact.getType() + "'.");
144         }
145       }
146 
147       File fatLib = resolveFatLib(project, xcodeConfiguration, mainArtifact);
148 
149       if (thinLibs.values().contains(null)) {
150 
151         if (fatLib != null) {
152           provideFatLib(fatLib, project, xcodeConfiguration, mainArtifact, sdks);
153         }
154         else {
155           throw new XCodeException("Neither all thin libs nor fat lib available for " + mainArtifact.getId() + ".");
156         }
157       }
158       else {
159 
160         if (preferFatLibs && fatLib != null)
161           provideFatLib(fatLib, project, xcodeConfiguration, mainArtifact, sdks);
162         else
163           provideThinLibs(thinLibs, xcodeConfiguration, mainArtifact, project);
164       }
165     }
166 
167     try {
168       prepareBundles(project, mainArtifact);
169     }
170     catch (SideArtifactNotFoundException e) {
171       LOGGER.info("Bundle not found for: '" + mainArtifact.getGroupId() + ":" + mainArtifact.getArtifactId() + ":"
172             + mainArtifact.getVersion() + ":" + mainArtifact.getType() + "'.");
173     }
174   }

176   private static void prepareRootFolders(MavenProject project, Set<String> configurations, Set<String> sdks)
177         throws IOException
178   {
179     mkdirs(FolderLayout.getFolderForExtractedMainArtifact(project));
180     mkdirs(FolderLayout.getFolderForExtractedBundles(project));
181     mkdirs(FolderLayout.getFolderForExtractedFrameworks(project));
182 
183     for (String configuration : configurations) {
184 
185       mkdirs(FolderLayout.getFolderForExtractedFatLibs(project, configuration));
186 
187       for (String sdk : sdks) {
188         mkdirs(FolderLayout.getFolderForExtractedHeaders(project, configuration, sdk));
189         mkdirs(FolderLayout.getFolderForExtractedLibs(project, configuration, sdk));
190       }
191     }

194   private File resolveThinLib(MavenProject project, final String xcodeConfiguration, final String sdk,
195         final Artifact primaryArtifact)
196   {

218   private void prepareBundles(MavenProject project, final Artifact primaryArtifact) throws MojoExecutionException,
219         SideArtifactNotFoundException, IOException
220   {
221     List<String> bundles = readBundleInformation(project, primaryArtifact);
222     if (bundles == null)

224

230   private void prepareBundle(MavenProject project, final Artifact primaryArtifact, String coords)
231         throws SideArtifactNotFoundException, MojoExecutionException
232   {
233     Artifact bundleArtifact = GAVUtil.getArtifact(coords);
234 
235     final org.sonatype.aether.artifact.Artifact artifact = downloadManager.resolveSideArtifact(bundleArtifact);
236 
237     if (artifact != null) {
238       final File source = artifact.getFile();
239 
240       final File target = new File(FolderLayout.getFolderForExtractedBundlesWithGA(project,
241             primaryArtifact.getGroupId(), primaryArtifact.getArtifactId()), bundleArtifact.getClassifier().replaceAll(
242             "~", File.separator)
243             + ".bundle");
244 
245       createDirectory(target);
246       com.sap.prd.mobile.ios.mios.FileUtils.unarchive(archiverManager, "zip", source, target);
247 
248       LOGGER.info("Bundle unarchived from " + source + " to " + target);
249 
250     }
251   }

253   @SuppressWarnings("unchecked")
254   private List<String> readBundleInformation(MavenProject project, Artifact primaryArtifact) throws IOException
255   {
256 
257     final File mainArtifactExtracted = FolderLayout.getFolderForExtractedPrimaryArtifact(project,
258           primaryArtifact);
259 
260     if (mainArtifactExtracted.exists())
261       com.sap.prd.mobile.ios.mios.FileUtils.deleteDirectory(mainArtifactExtracted);
262 
263     if (!mainArtifactExtracted.mkdirs())
264       throw new IOException("Cannot create directory for expanded mainartefact of " + primaryArtifact.getGroupId()
265             + ":" + primaryArtifact.getArtifactId() + " (" + mainArtifactExtracted + ").");
266 
267     com.sap.prd.mobile.ios.mios.FileUtils.unarchive(archiverManager, "tar", primaryArtifact.getFile(),
268           mainArtifactExtracted);
269 
270     LOGGER.info("Main artifact extracted to '" + mainArtifactExtracted + "'.");
271 
272     File bundleFile = new File(mainArtifactExtracted, "bundles.txt");
273     if (!bundleFile.exists())
274       return null;
275 
276     try {
277       return FileUtils.readLines(bundleFile);
278     }
279     catch (IOException e) {
280       throw new RuntimeException(e);
281     }
282

285   private void provideThinLib(final File source, MavenProject project, final String xcodeConfiguration,
286         final String sdk, final Artifact primaryArtifact) throws IOException
287   {

290           sdk,
291           primaryArtifact.getGroupId(), primaryArtifact.getArtifactId()), getArchiveFileName(primaryArtifact));

314   private void provideFatLib(File source, MavenProject project, final String xcodeConfiguration,
315         final Artifact primaryArtifact, Set<String> sdks) throws IOException
316   {
317 
318     final File target = new File(FolderLayout.getFolderForExtractedFatLibsWithGA(project, xcodeConfiguration,
319           primaryArtifact.getGroupId(), primaryArtifact.getArtifactId()), getArchiveFileName(primaryArtifact));
320 
321     if (ArtifactUtils.isSnapshot(primaryArtifact.getVersion()) || !useSymbolicLinks()) {
322       FileUtils.copyFile(source, target);
323     }
324     else {
325       com.sap.prd.mobile.ios.mios.FileUtils.createSymbolicLink(source, target);
326     }
327 
328     final FatLibAnalyzer lipoHelper = new FatLibAnalyzer(target);
329 
330     //
331     // TODO improve hard coded strings for iphoneos and iphonesimulator below
332     //
333     if (sdks.contains("iphoneos")) {
334       if (!lipoHelper.containsArmv())
335         LOGGER.warning("Fat library '" + lipoHelper.getFatLibrary() + "' does not contain a library for armv*.");
336       else
337         LOGGER.info("Fat library '" + lipoHelper.getFatLibrary() + "'contains a library for armv*.");
338     }
339     else if (sdks.contains("iphonesimulator")) {
340       if (!lipoHelper.containsI386())
341         LOGGER.warning("Fat library '" + lipoHelper.getFatLibrary() + "' does not contain a library for i386.");
342       else
343         LOGGER.info("Fat library '" + lipoHelper.getFatLibrary() + "'contains a library for i386.");
344     }

352   private void prepareHeaders(MavenProject project, String xcodeConfiguration,
353         final String sdk, final Artifact primaryArtifact) throws MojoExecutionException, SideArtifactNotFoundException
354   {
355 
356     final org.sonatype.aether.artifact.Artifact headersArtifact = downloadManager.resolveSideArtifact(primaryArtifact,
357           xcodeConfiguration + "-" + sdk, TYPE_HEADERS);
358 
359     if (headersArtifact != null) {

362             xcodeConfiguration, sdk,
363             primaryArtifact.getGroupId(), primaryArtifact.getArtifactId());
364 
365       createDirectory(headersDirectory);

367       extractHeaders(headersDirectory, headersArtifact.getFile());

371   private void prepareFramework(MavenProject project, final Artifact primaryArtifact, Collection<String> configurations)
372         throws MojoExecutionException
373   {
374 
375     for (String configuration : configurations) {
376       try {
377         org.sonatype.aether.artifact.Artifact frameworkArtifact = downloadManager.resolveSideArtifact(primaryArtifact,
378               configuration, Types.FRAMEWORK);
379         extractFramework(project, primaryArtifact, configuration, frameworkArtifact.getFile());
380       }
381       catch (SideArtifactNotFoundException e) {
382         LOGGER
383           .warning("Framework '"
384                 + primaryArtifact
385                 + "' does not contain configuration specific variant. Will download the generic framework for configuration '"
386                 + configuration + "'.");
387         handlePrimaryArtifact(project, primaryArtifact, configurations);
388       }
389     }
390   }

392   private void extractFramework(MavenProject project, final Artifact primaryArtifact, String configuration,
393         File frameworkArtifact) throws MojoExecutionException
394   {
395     File target = FolderLayout.getFolderForExtractedFrameworkswithGA(project, primaryArtifact.getGroupId(),
396           primaryArtifact.getArtifactId(), configuration);
397     createDirectory(target);
398     try {
399       extractFileWithShellScript(frameworkArtifact, target, new File(project.getBuild().getDirectory()));
400     }
401     catch (IOException ioe) {
402       throw new MojoExecutionException("Cannot unarchive framework from " + frameworkArtifact + " to "
403             + target);
404     }
405   }

407   private void handlePrimaryArtifact(MavenProject project, final Artifact primaryArtifact,
408         Collection<String> configurations)
409         throws MojoExecutionException
410   {
411     if (primaryArtifact != null) {
412       final File source = primaryArtifact.getFile();
413       for (String configuration : configurations) {
414         final File target = FolderLayout.getFolderForExtractedFrameworkswithGA(project,
415               primaryArtifact.getGroupId(), primaryArtifact.getArtifactId());
416         extractFramework(project, primaryArtifact, configuration, source);
417 
418         try {
419           extractFileWithShellScript(source, target, new File(project.getBuild().getDirectory()));
420         }
421         catch (IOException ioe) {
422           throw new MojoExecutionException("Cannot unarchive framework from " + source + " to " + target);
423         }
424 
425         LOGGER.info("Framework unarchived from " + source + " to " + target);
426       }
427     }

447   {
448

460   private void extractFileWithShellScript(File sourceFile, File destinationFolder, File tmpFolder) throws IOException
461   {
462     File workingDirectory = new File(tmpFolder, "scriptWorkingDir");
463     workingDirectory.deleteOnExit();
464     ScriptRunner.copyAndExecuteScript(System.out, "/com/sap/prd/mobile/ios/mios/unzip.sh", workingDirectory,
465           sourceFile.getCanonicalPath(),
466           destinationFolder.getCanonicalPath());
467   }
