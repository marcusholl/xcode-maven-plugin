39    {
40  
41      return new File(buildDir, configuration + "-" + sdk + "/lib" + projectName + ".a");
42    }
43  
44    static File getBundleDirectory(final File srcDir, final String bundleName)
45    {
46  
47      return new File(srcDir, bundleName + ".bundle");
48    }
49  
50    // TODO invent better method name
51    static File getAppFolder(final File baseDirectory, final String configuration, final String sdk)
52    {
53      return new File(getBuildDir(baseDirectory), configuration + "-" + sdk);
54    }
55  
56    static File getBuildDir(final File baseDirectory)
57    {
58  
59      if (baseDirectory == null)
60        // exception here is required since File constructor below
61        // does not fail when called with null value for baseDirectory.
62        // We would not be acting fail fast without throwing an exception here.
63        throw new IllegalStateException("baseDirectory was null");
64      return new File(baseDirectory, "build");
65    }
