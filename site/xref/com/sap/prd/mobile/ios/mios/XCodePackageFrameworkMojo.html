32  /**
33   * Packages the framework built by Xcode and prepares the generated artifact for deployment.
34   * 
35   * @goal package-framework
36   * 
37   */
38  public class XCodePackageFrameworkMojo extends BuildContextAwareMojo
39  {
40    /**
41     * The frameworks are built for all defined configurations, i.e Debug and Release. The parameter
42     * here defines the configuration, used for the primary artifact generation.
43     * 
44     * @parameter expression="${xcode.primaryFmwkConfiguration}" default-value="Release"
45     * @since 1.4.2
46     */
47    private String primaryFmwkConfiguration;
48  
49    /**
50     * 
51     * @parameter expression="${xcode.primaryFmwkSdk}" default-value="iphoneos"
52     * @since 1.9.4
53     */
54    private String primaryFmwkSdk;
55  
56    /**
57     * @component
58     */
59    private MavenProjectHelper projectHelper;
60  
61    @Override
62    public void execute() throws MojoExecutionException, MojoFailureException
63    {
64  
65      final File mainArtifact = getFrameworkArtifact(getPrimaryFmwkSdk(), getPrimaryFmwkConfiguration());
66      project.getArtifact().setFile(mainArtifact);
67  
68      for (String configuration : getConfigurations()) {
69        final File frameworkArtifact = getFrameworkArtifact(getPrimaryFmwkSdk(), configuration);
70  
71        projectHelper.attachArtifact(project, frameworkArtifact, configuration);
72        getLog().info("Artifact file '" + frameworkArtifact + "' attached for " + project.getArtifact());
73      }
74    }
75  
76    private File getFrameworkArtifact(final String sdk, String configuration) throws MojoExecutionException
77    {
78      String builtProductsDirName = null;
79      String productName = getProductName(configuration, sdk);
80      try {
81        builtProductsDirName = EffectiveBuildSettings.getBuildSetting(
82              getXCodeContext(XCodeContext.SourceCodeLocation.WORKING_COPY, configuration, sdk),
83              EffectiveBuildSettings.BUILT_PRODUCTS_DIR);
84      }
85      catch (XCodeException ex) {
86        throw new MojoExecutionException(ex.getMessage(), ex);
87      }
88  
89      File builtProductsDir = new File(builtProductsDirName);
90      String fmwkDirName = productName + ".framework";
91      File fmwkDir = new File(builtProductsDir, fmwkDirName);
92  
93      validateFrmkStructure(fmwkDir);
94  
95      String artifactName = productName + "." + Types.FRAMEWORK;
96      zipFmwk(builtProductsDir, artifactName, fmwkDirName);
97  
98      File frameworkArtifact = new File(builtProductsDir, artifactName);
99      return frameworkArtifact;
100   }
101 
102   private void zipFmwk(File workingDirectory, String artifactName, String zipDirName) throws MojoExecutionException
103   {
104     try {
105       String[] zipCmd = new String[] { "zip", "-r", "-y", "-q", artifactName, zipDirName, zipDirName.replaceAll("framework", "embeddedframework") };
106       getLog().info("Executing: " + StringUtils.join(zipCmd, ' '));
107       int exitCode = Forker.forkProcess(System.out, workingDirectory, zipCmd);
108       if (exitCode != 0) {
109         throw new MojoExecutionException("Could not package the Xcode framework.");
110       }
111     }
112     catch (IOException e) {
113       throw new MojoExecutionException("Could not package the Xcode framework.", e);
114     }
115   }
116 
117   private void validateFrmkStructure(File fmwkDir) throws MojoExecutionException
118   {
119     FrameworkStructureValidator fmwkValidator = new FrameworkStructureValidator(fmwkDir);
120 
121     List<String> validationErrors = fmwkValidator.validate();
122     if (!validationErrors.isEmpty()) {
123       throw new MojoExecutionException("The validation of the built framework '" + fmwkDir.getAbsolutePath()
124             + "' failed: " + validationErrors);
125     }
126   }
127 
128   protected String getPrimaryFmwkConfiguration() throws MojoExecutionException
129   {
130     if (getConfigurations().contains(primaryFmwkConfiguration)) {
131       return primaryFmwkConfiguration;
132     }
133     throw new MojoExecutionException("The primaryFmwkConfiguration '" + primaryFmwkConfiguration
134           + "' is not part of the configurations '" + getConfigurations() + "' defined for this project");
135   }
136 
137   protected String getPrimaryFmwkSdk() throws MojoExecutionException
138   {
139     if (getSDKs().contains(primaryFmwkSdk)) {
140       return primaryFmwkSdk;
141     }
142     throw new MojoExecutionException("The primaryFmwkSdk '" + primaryFmwkSdk
143           + "' is not part of the sdks '" + getSDKs() + "' defined for this project");
144   }
