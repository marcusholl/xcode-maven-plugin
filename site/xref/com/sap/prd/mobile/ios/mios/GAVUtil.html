31  // Format:  <groupId>:<artifactId>[:<extension>[:<classifier>]]:<version>
32  class GAVUtil
33  {
34    static String toColonNotation(String groupId, String artifactId, String version, String extension, String classifier)
35    {
36      // For our use case we expect all parameter to be not null (or empty).
37      // Our use case is handling side artifacts with requires a classifier an an extension.
38      ensureNotNullOrEmpty(groupId, GROUP_ID);
39      ensureNotNullOrEmpty(artifactId, ARTIFACT_ID);
40      ensureNotNullOrEmpty(version, VERSION);
41      ensureNotNullOrEmpty(extension, EXTENSION);
42      ensureNotNullOrEmpty(classifier, CLASSIFIER);
43  
44      return groupId + ":" + artifactId + ":" + extension + ":" + classifier + ":" + version;
45    }
46  
47    private static void ensureNotNullOrEmpty(String str, String parameterName)
48    {
49      if (str == null || str.isEmpty())
50        throw new IllegalStateException("Value of parameter '" + parameterName + "' was null or empty.");
51  
52    }

54    static org.apache.maven.artifact.DefaultArtifact getArtifact(String coords)
55    {
56      DefaultArtifact aetherArtifact = new DefaultArtifact(coords);
57      return new org.apache.maven.artifact.DefaultArtifact(aetherArtifact.getGroupId(), aetherArtifact.getArtifactId(),
58            aetherArtifact.getVersion(), null, aetherArtifact.getExtension(), aetherArtifact.getClassifier(), null) {
59  
60        // The DefaultArtifact created here is only used internally as a
61        // vehicle. Since we do not set a scope and an artifact handler
62        // these methods cannot work in a reasonable way. Instead returning
63        // null which might lead to problems later on we throw an
64        // exception right when the corresponding methods are called.
65  
66        @Override
67        public String getScope()
68        {
69          throw new UnsupportedOperationException();
70        }
71  
72        @Override
73        public ArtifactHandler getArtifactHandler()
74        {
75          throw new UnsupportedOperationException();
76        }
77      };
78    }
